// النطاق العالمي الموحد لسيرفر Render الخاص بك const API_BASE_URL = "https://smart-fleet-backend-h6pc.onrender.com"; let map, marker; let watchId = null; let trackingShipment = null; // دالة استخراج الـ ID مالت الشحنة من الرابط function getTrackIdFromURL() { const params = new URLSearchParams(window.location.search); return params.get('id'); } document.addEventListener("DOMContentLoaded", async function() { const trackingId = getTrackIdFromURL(); if (!trackingId) { document.getElementById('status-text').innerText = "❌ خطأ: رقم التتبع غير موجود بالرابط!"; return; } document.getElementById('track-id-display').innerText = trackingId; // تهيئة الخريطة الجغرافية map = L.map('track-map').setView([33.3152, 44.3661], 14); L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map); marker = L.marker([33.3152, 44.3661]).addTo(map); // جلب بيانات الشحنة من قاعدة البيانات السحابية فوراً await fetchShipmentDetails(trackingId); // تحديث تلقائي للبيانات كل 5 ثوانٍ لمراقبة الحالة setInterval(() => { fetchShipmentDetails(trackingId); }, 5000); }); // جلب تفاصيل الطرد والكابتن من سيرفر Render async function fetchShipmentDetails(trackingId) { try { const res = await fetch(`${API_BASE_URL}/api/shipments`); const shipments = await res.json(); const order = shipments.find(s => s.tracking_id === trackingId); if (!order) { document.getElementById('status-text').innerText = "الشحنة غير مسجلة بالنظام."; return; } trackingShipment = order; document.getElementById('status-text').innerText = order.status; document.getElementById('merchant-name').innerText = order.sender; document.getElementById('customer-zone').innerText = order.address; // جلب إحداثيات الكابتن المكلف بالطرد const driversRes = await fetch(`${API_BASE_URL}/api/drivers`); const drivers = await driversRes.json(); const driver = drivers.find(d => d.id === order.driver_id); if (driver) { document.getElementById('driver-name').innerText = driver.name; document.getElementById('driver-car').innerText = driver.vehicle; const lat = Number(driver.lat); const lng = Number(driver.lng); // تحريك النقطة على الخريطة حسب بث الـ GPS الحقيقي للكابتن marker.setLatLng([lat, lng]); map.setView([lat, lng], map.getZoom()); marker.bindPopup(`موقع الكابتن الحالي
الحالة: ${driver.status}`).openPopup(); } } catch (err) { console.error("Error loading tracking data:", err); } } // ================= محرك بث الـ GPS الحي من الهاتف إلى السيرفر العالمي ================= async function toggleLiveGPS() { const btn = document.getElementById('gps-broadcast-btn'); const driverId = trackingShipment ? trackingShipment.driver_id : 101; // التعيين الافتراضي أو كابتن الشحنة if (watchId !== null) { // إيقاف البث navigator.geolocation.clearWatch(watchId); watchId = null; btn.innerHTML = "📍 بث موقعي الفعلي الآن عبر الـ GPS لايف"; btn.style.background = "var(--primary)"; alert("تم إيقاف بث الموقع الجغرافي السحابي."); return; } if (!navigator.geolocation) { alert("متصفحك القديم لا يدعم مستشعرات الـ GPS الحقيقية!"); return; } // تشغيل بث المستشعرات الفعلي لا سلكياً وبأعلى دقة watchId = navigator.geolocation.watchPosition( async (position) => { const lat = position.coords.latitude; const lng = position.coords.longitude; const speed = Math.floor(position.coords.speed * 3.6) || 0; // تحويل لـ كم/س console.log(`📡 GPS Update Node Sent: ${lat}, ${lng} | Speed: ${speed}`); try { // إرسال الإحداثيات الحقيقية فوراً عبر الإنترنت لتستقر في Neon DB await fetch(`${API_BASE_URL}/api/drivers/${driverId}/gps`, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ lat: lat, lng: lng, speed: speed, is_real_gps: true }) }); } catch (err) { console.error("Failed to stream GPS node to Render:", err); } }, (error) => { alert("فشل التقاط بث الـ GPS! تأكد من تفعيل الموقع في هاتفك ومنح الصلاحية."); }, { enableHighAccuracy: true, maximumAge: 0, timeout: 5000 } ); btn.innerHTML = "📡 جاري البث الحي لايف إلى السحاب... اضغط للإيقاف"; btn.style.background = "#ff0055"; alert("تم تفعيل شريحة البث الحي بنجاح! ستقفز النقطة الآن في لوحة التحكم باللابتوب فوق موقعك الحقيقي."); }
